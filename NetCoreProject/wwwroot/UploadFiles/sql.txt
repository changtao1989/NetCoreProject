SELECT t2.SCZSBH,t2.SCGGDH,t2.SONSCGGDH,t3.WLBM,t5.SCCJJQ,t2.SCZSSL
FROM dbo.T_TSData AS t1
INNER JOIN dbo.sct27 AS t2 ON t2.SCGGDH = t1.Scggdh AND t2.SONSCGGDH = t1.SonScggdh
INNER JOIN dbo.sct26 AS t5 ON t5.SCZSBH = t2.SCZSBH
INNER JOIN dbo.T_BLData_PLD AS t3 ON t3.SCGGDH = t1.Scggdh
WHERE PLStatus=1 AND CCStatus=1 AND TLStatus=0 AND MESStatus=10 AND WMSStatus=20 AND t3.child_sku IS NULL AND t3.WLType='F'
